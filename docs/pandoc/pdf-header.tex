\usepackage{fontspec}
\setmonofont{DejaVu Sans Mono}[Scale=0.85]

\usepackage{fancyhdr}
\usepackage{needspace}
\usepackage{etoolbox}
\usepackage{titlesec}
\usepackage{graphicx}

% Scale images to fit within text area (centered via Pandoc's figure environment)
\makeatletter
\def\maxwidth{\ifdim\Gin@nat@width>\linewidth\linewidth\else\Gin@nat@width\fi}
\def\maxheight{\ifdim\Gin@nat@height>0.85\textheight 0.85\textheight\else\Gin@nat@height\fi}
\makeatother
\setkeys{Gin}{width=\maxwidth,height=\maxheight,keepaspectratio}

% ## part titles — Large, centered (Pandoc maps ## → \subsection)
\titleformat{\subsection}[block]{\centering\normalfont\Large\bfseries}{}{0em}{}
\titlespacing*{\subsection}{0pt}{1.2em}{0.8em}

% ### question titles — large, bold (Pandoc maps ### → \subsubsection)
\titleformat{\subsubsection}[block]{\normalfont\large\bfseries}{}{0em}{}
\titlespacing*{\subsubsection}{0pt}{0pt}{0.6em}

\usepackage{fvextra}
\usepackage{mdframed}

% Footer page numbers only; no print header chrome
\pagestyle{fancy}
\fancyhead{}
\fancyfoot[C]{\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% Title page → TOC → body on separate pages
\pretocmd{\tableofcontents}{\newpage}{}{}
\apptocmd{\tableofcontents}{\newpage}{}{}

% Each ## part starts on a new page (after the preceding question ends); first after TOC skips extra break
\newif\iffirstsubsection
\firstsubsectiontrue
\newif\iffirstsubsubsection
\firstsubsubsectiontrue
\pretocmd{\subsection}{%
  \iffirstsubsection
    \global\firstsubsectionfalse
  \else
    \clearpage
  \fi
  \global\firstsubsubsectiontrue
}{}{}
\pretocmd{\subsubsection}{%
  \iffirstsubsubsection
    \global\firstsubsubsectionfalse
  \else
    \clearpage
    \Needspace{8\baselineskip}
  \fi
}{}{}

% Keep code blocks on one page when they fit
\BeforeBeginEnvironment{Shaded}{\Needspace{10\baselineskip}}
\BeforeBeginEnvironment{verbatim}{\Needspace{10\baselineskip}}

% Override Pandoc defaults after template loads (header-includes run before Highlighting is defined)
\AtBeginDocument{%
  % Underline #### section headings (Pandoc maps #### → \paragraph)
  \makeatletter
  \renewcommand\paragraph[1]{%
    \Needspace{6\baselineskip}%
    \par\addvspace{0.8ex}%
    \noindent{\normalfont\normalsize\bfseries\underline{#1}}%
    \par\addvspace{0.4ex}%
  }
  \makeatother
  \DefineVerbatimEnvironment{Highlighting}{Verbatim}{%
    breaklines=true,
    breakanywhere=true,
    commandchars=\\\{\}%
  }%
  \DefineVerbatimEnvironment{verbatim}{Verbatim}{%
    breaklines=true,
    breakanywhere=true
  }%
  \renewenvironment{Shaded}{%
    \begin{mdframed}[
      linewidth=0.4pt,
      linecolor=gray!60,
      backgroundcolor=white,
      innerleftmargin=6pt,
      innerrightmargin=6pt,
      innertopmargin=6pt,
      innerbottommargin=6pt,
      skipabove=0.4em,
      skipbelow=0.4em
    ]%
  }{%
    \end{mdframed}%
  }%
}
