\def\maxwidth{\ifdim\Gin@nat@width>\linewidth\linewidth\else\Gin@nat@width\fi}

% Document title at top of TOC page; body starts on a new page after TOC
\makeatletter
\renewcommand{\maketitle}{%
  \begingroup
  \centering
  \normalfont
  \vspace*{0.5em}
  {\LARGE\bfseries\@title\par}
  \vspace{1.5em}
  \endgroup
}
\makeatother
\apptocmd{\tableofcontents}{\newpage}{}{}

% Roman numerals in TOC as "IX. Section Title" (inline, bold); body titles stay unnumbered
\makeatletter
\newcommand{\dh@toc@subsection@numberline}[1]{#1.\ }
\AtBeginDocument{%
  \setcounter{secnumdepth}{2}%
  \renewcommand{\thesubsection}{\Roman{subsection}}%
}
\renewcommand*\l@subsection{\@dottedtocline{2}{0em}{0em}}%
\let\dh@l@subsection\l@subsection
\renewcommand*\l@subsection[2]{%
  \begingroup
  \bfseries
  \let\numberline\dh@toc@subsection@numberline
  \dh@l@subsection{#1}{#2}%
  \endgroup
}
\renewcommand{\contentsname}{Table of Contents}
% ### questions in TOC (1., 2., …) — less indent than default 3.8em
\renewcommand*\l@subsubsection{\@dottedtocline{3}{1.5em}{0em}}
\makeatother

% Each ## part starts on a new page (after the preceding question ends); first after TOC skips extra break
\newif\iffirstsubsection
\firstsubsectiontrue
\newif\iffirstsubsubsection
\firstsubsubsectiontrue
\pretocmd{\subsection}{%
  \iffirstsubsection
    \global\firstsubsectionfalse
  \else
    \clearpage
  \fi
  \global\firstsubsubsectiontrue
}{}{}
\pretocmd{\subsubsection}{%
  \iffirstsubsubsection
    \global\firstsubsubsectionfalse
  \else
    \clearpage
    \Needspace{8\baselineskip}
  \fi
}{}{}

% Keep code blocks on one page when they fit
\BeforeBeginEnvironment{Shaded}{\Needspace{10\baselineskip}}
\BeforeBeginEnvironment{verbatim}{\Needspace{10\baselineskip}}

% Override Pandoc defaults after template loads (header-includes run before Highlighting is defined)
\AtBeginDocument{%
  % Underline #### section headings (Pandoc maps #### → \paragraph)
  \makeatletter
  \renewcommand\paragraph[1]{%
    \Needspace{3\baselineskip}%
    \par\addvspace{0.8ex}%
    \noindent{\normalfont\normalsize\bfseries\underline{#1}}%
    \par\addvspace{0.4ex}%
  }
  \makeatother
  \DefineVerbatimEnvironment{Highlighting}{Verbatim}{%
    breaklines=true,
    breakanywhere=true,
    commandchars=\\\{\}%
  }%
  \DefineVerbatimEnvironment{verbatim}{Verbatim}{%
    breaklines=true,
    breakanywhere=true
  }%
  \renewenvironment{Shaded}{%
    \begin{mdframed}[
      linewidth=0.4pt,
      linecolor=gray!60,
      backgroundcolor=white,
      innerleftmargin=6pt,
      innerrightmargin=6pt,
      innertopmargin=6pt,
      innerbottommargin=6pt,
      skipabove=0.4em,
      skipbelow=0.4em
    ]%
  }{%
    \end{mdframed}%
  }%
}
