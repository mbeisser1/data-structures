\usepackage{fontspec}
\setmonofont{DejaVu Sans Mono}[Scale=0.85]

\usepackage{fancyhdr}
\usepackage{needspace}
\usepackage{etoolbox}
\usepackage{titlesec}

% Center each ## part title (Pandoc maps ## → \subsection)
\titleformat{\subsection}[block]{\centering\normalfont\large\bfseries}{}{0em}{}
\usepackage{fvextra}
\usepackage{mdframed}

% Footer page numbers only; no print header chrome
\pagestyle{fancy}
\fancyhead{}
\fancyfoot[C]{\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% Title page → TOC → body on separate pages
\pretocmd{\tableofcontents}{\newpage}{}{}
\apptocmd{\tableofcontents}{\newpage}{}{}

% Each ## part title starts on a new page (Pandoc maps ## → \subsection; skip first after TOC)
\newif\iffirstsubsection
\firstsubsectiontrue
\pretocmd{\subsection}{%
  \iffirstsubsection
    \global\firstsubsectionfalse
  \else
    \clearpage
  \fi
}{}{}

% Each ### question starts on a new page (Pandoc maps ### → \subsubsection)
\pretocmd{\subsubsection}{\clearpage\Needspace{8\baselineskip}}{}{}
\pretocmd{\paragraph}{\Needspace{6\baselineskip}}{}{}

% Keep code blocks on one page when they fit
\BeforeBeginEnvironment{Shaded}{\Needspace{10\baselineskip}}
\BeforeBeginEnvironment{verbatim}{\Needspace{10\baselineskip}}

% Override Pandoc defaults after template loads (header-includes run before Highlighting is defined)
\AtBeginDocument{%
  \DefineVerbatimEnvironment{Highlighting}{Verbatim}{%
    breaklines=true,
    breakanywhere=true,
    commandchars=\\\{\}%
  }%
  \DefineVerbatimEnvironment{verbatim}{Verbatim}{%
    breaklines=true,
    breakanywhere=true
  }%
  \renewenvironment{Shaded}{%
    \begin{mdframed}[
      linewidth=0.4pt,
      linecolor=gray!60,
      backgroundcolor=white,
      innerleftmargin=6pt,
      innerrightmargin=6pt,
      innertopmargin=6pt,
      innerbottommargin=6pt,
      skipabove=0.4em,
      skipbelow=0.4em
    ]%
  }{%
    \end{mdframed}%
  }%
}

\setlength{\parskip}{0.5em plus 0.1em minus 0.1em}
